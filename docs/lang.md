# declare a const

# define a function

    return x + y
}

# define a struct
struct Student {
    var name : string
    var age : int
}

# main function
function main(argv:[string]) : int {
    # print values
    print("Hi!",1,2)

    # if statement
    if x == 1 {
        print(1)
    }else if x == 2{
        print(2)
    }else {
        print(3)
    }

    # for statement
    for x = 1 to 10 {
        print(x)
    }

    # while statement
    while x < 100 {
        x = x + 1
        print(x)
    }

    # call a function
    my_func("arg 1",1)

    # access to struct members
    var john : Student
    john.name = "John"
    john.age = 23
    

    return 0
}
